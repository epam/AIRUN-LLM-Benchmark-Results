import React, { useEffect, useState } from 'react';
import { useAppDispatch, useAppSelector } from '../app/hooks';
import { setPage } from '../features/page/pageSlice';
import axios from 'axios';

interface PageFormProps {}

const PageForm: React.FC<PageFormProps> = () => {
  const dispatch = useAppDispatch();
  const page = useAppSelector((state) => state.page);
  const [newerVersion, setNewerVersion] = useState(false);

  useEffect(() => {
    const elements = ['title', 'description', 'publish', 'scheduleDate', 'url'];
    if (window.location.pathname !== '/new') {
      let hasNewerVersion = false;
      elements.forEach((value) => {
        if (
          localStorage.getItem(window.location.pathname + value) !==
            page[value as keyof typeof page] &&
          localStorage.getItem(window.location.pathname + value) !== 'null'
        ) {
          hasNewerVersion = true;
        }
      });
      setNewerVersion(hasNewerVersion);
    }
  }, [page]);

  const handleLocalVersion = () => {
    const elements = ['title', 'description', 'publish', 'scheduleDate', 'url'];
    elements.forEach((value) => {
      if (localStorage.getItem(window.location.pathname + value) !== 'null') {
        dispatch(
          setPage({ ...page, [value]: localStorage.getItem(window.location.pathname + value) })
        );
        localStorage.setItem(window.location.pathname + value, null);
      }
    });
    setNewerVersion(false);
  };

  const handleDeleteNewerVersion = () => {
    const elements = ['title', 'description', 'publish', 'scheduleDate', 'url'];
    elements.forEach((value) => {
      localStorage.setItem(window.location.pathname + value, null);
    });
    setNewerVersion(false);
  };

  const handleTitleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    dispatch(setPage({ ...page, title: e.target.value }));
  };

  const handleDescriptionChange = (e: React.ChangeEvent<HTMLTextAreaElement>) => {
    dispatch(setPage({ ...page, description: e.target.value }));
  };

  const handleUrlChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    dispatch(setPage({ ...page, url: e.target.value }));
  };

  const handleTypeChange = (e: React.ChangeEvent<HTMLSelectElement>) => {
    dispatch(setPage({ ...page, type: e.target.value }));
  };

  const handleSavePage = () => {
    // Implement save logic
  };

  return (
    <div>
      {newerVersion && (
        <div>
          <p>Newer version available</p>
          <button onClick={handleDeleteNewerVersion}>Discard</button>
          <button onClick={handleLocalVersion}>Use Local Version</button>
        </div>
      )}
      <div>
        <input value={page.title} onChange={handleTitleChange} placeholder="Title" />
        <textarea value={page.description} onChange={handleDescriptionChange} placeholder="Description" />
        <input value={page.url} onChange={handleUrlChange} placeholder="URL" />
        <select value={page.type} onChange={handleTypeChange}>
          <option value="type1">Type 1</option>
          <option value="type2">Type 2</option>
        </select>
        <button onClick={handleSavePage}>Save</button>
      </div>
    </div>
  );
};

export default PageForm;
