## ✅ 6. Local Storage Hook

### hooks/useLocalStorage.ts

```ts
import { useState } from 'react';

export function useLocalStorage<T>(key: string, initialValue: T) {
  const [storedValue, setStoredValue] = useState<T>(() => {
    const item = localStorage.getItem(key);
    return item ? JSON.parse(item) : initialValue;
  });

  const setValue = (value: T) => {
    setStoredValue(value);
    localStorage.setItem(key, JSON.stringify(value));
  };

  const removeValue = () => {
    localStorage.removeItem(key);
  };

  return [storedValue, setValue, removeValue] as const;
}
```

## ✅ 7. Page Form Component

### features/page/PageForm.tsx

```tsx
import React, { useEffect, useState } from 'react';
import { useAppDispatch, useAppSelector } from '../../app/hooks';
import { setPage } from './pageSlice';
import { REST } from '../rest/restAPI';
import { useLocalStorage } from '../../hooks/useLocalStorage';
import { useTranslation } from 'react-i18next';
import { useNavigate, useParams } from 'react-router-dom';

const PageForm: React.FC = () => {
  const dispatch = useAppDispatch();
  const page = useAppSelector((state) => state.page);
  const { t } = useTranslation();
  const navigate = useNavigate();
  const { url } = useParams<{ url: string }>();

  const [title, setTitle] = useState(page.title);
  const [description, setDescription] = useState(page.description);
  const [tags, setTags] = useState<string[]>(page.tags || []);
  const [suggestions, setSuggestions] = useState<string[]>([]);
  const [autoURL, setAutoURL] = useState(true);
  const [urlValue, setUrlValue] = useState(page.url);

  const [storedTitle, setStoredTitle] = useLocalStorage(`${url}title`, '');
  const [storedDescription, setStoredDescription] = useLocalStorage(`${url}description`, '');

  useEffect(() => {
    dispatch(setPage({ title, description, url: urlValue, tags }));
  }, [title, description, urlValue, tags]);

  const handleTitleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const value = e.target.value;
    setTitle(value);
    if (autoURL) {
      const generatedUrl = value.toLowerCase().replace(/ /g, '-').replace(/[^\w-]+/g, '');
      setUrlValue(generatedUrl);
    }
  };

  const handleTagChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const tagList = e.target.value.split(',').map((tag) => tag.trim());
    setTags(tagList);
    const lastTag = tagList[tagList.length - 1];
    if (lastTag) {
      REST.contentTags
        .query(lastTag)
        .then((res) => setSuggestions(res.data))
        .catch(() => setSuggestions([]));
    }
  };

  const handleSuggestionClick = (tag: string) => {
    const updatedTags = [...tags.slice(0, -1), tag, ''];
    setTags(updatedTags);
    setSuggestions([]);
  };

  const handleSave = async () => {
    if (!title || !urlValue) {
      alert(t('page_no_url'));
      return;
    }

    try {
      const response = await REST.content.save({
        title,
        description,
        url: urlValue,
        type: page.type,
        published: page.published,
        published_date: new Date().toISOString(),
        author: '1', // Replace with actual user ID
      });

      navigate(`/${urlValue}`);
    } catch (error) {
      alert(t('page_error_saving'));
    }
  };

  return (
    <div className="pg-editor form-case">
      <label>{t('title')}</label>
      <input type="text" value={title} onChange={handleTitleChange} />

      <label>{t('description')}</label>
      <textarea value={description} onChange={(e) => setDescription(e.target.value)} />

      <label>{t('tags')}</label>
      <input type="text" value={tags.join(',')} onChange={handleTagChange} />
      <div className="tag-suggestions">
        {suggestions.map((tag) => (
          <div key={tag} onClick={() => handleSuggestionClick(tag)}>
            {tag}
          </div>
        ))}
      </div>

      <label>{t('url')}</label>
      <input type="text" value={urlValue} onChange={(e) => setUrlValue(e.target.value)} />

      <button onClick={handleSave}>{t('save')}</button>
    </div>
  );
};

export default PageForm;
```
