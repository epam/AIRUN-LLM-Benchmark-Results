```typescript
import { createSlice, PayloadAction } from '@reduxjs/toolkit';
import { nanoid } from 'nanoid';
import axios from 'axios';

interface MenuItem {
  id: string;
  title: string;
  url: string;
  items: MenuItem[];
}

interface MenuState {
  menus: MenuItem[];
  selectedMenu: MenuItem | null;
}

const initialState: MenuState = {
  menus: [],
  selectedMenu: null,
};

export const menuSlice = createSlice({
  name: 'menu',
  initialState,
  reducers: {
    setMenus: (state, action: PayloadAction<MenuItem[]>) => {
      state.menus = action.payload;
    },
    addMenu: (state, action: PayloadAction<{ name: string }>) => {
      const newMenu: MenuItem = {
        id: nanoid(),
        title: action.payload.name,
        url: '',
        items: [],
      };
      state.menus.push(newMenu);
    },
    deleteMenu: (state, action: PayloadAction<string>) => {
      state.menus = state.menus.filter(menu => menu.id !== action.payload);
    },
    selectMenu: (state, action: PayloadAction<string>) => {
      state.selectedMenu = state.menus.find(menu => menu.id === action.payload) || null;
    },
    updateMenu: (state, action: PayloadAction<MenuItem>) => {
      const index = state.menus.findIndex(menu => menu.id === action.payload.id);
      if (index !== -1) {
        state.menus[index] = action.payload;
      }
    },
  },
});

export const { setMenus, addMenu, deleteMenu, selectMenu, updateMenu } = menuSlice.actions;

export default menuSlice.reducer;
```

import React, { useEffect } from 'react';
import { useDispatch, useSelector } from 'react-redux';
import { RootState } from '../../app/store';
import { setMenus, addMenu, deleteMenu, selectMenu, updateMenu } from './menuSlice';
import axios from 'axios';

const MenuManager: React.FC = () => {
  const dispatch = useDispatch();
  const menus = useSelector((state: RootState) => state.menu.menus);
  const selectedMenu = useSelector((state: RootState) => state.menu.selectedMenu);

  useEffect(() => {
    axios.get('/api/menus').then(response => {
      dispatch(setMenus(response.data));
    });
  }, [dispatch]);

  const handleAddMenu = () => {
    dispatch(addMenu({ name: 'New Menu' }));
  };

  const handleDeleteMenu = (id: string) => {
    dispatch(deleteMenu(id));
  };

  const handleSelectMenu = (id: string) => {
    dispatch(selectMenu(id));
  };

  return (
    <div>
      <button onClick={handleAddMenu}>Add Menu</button>
      {menus.map(menu => (
        <div key={menu.id}>
          <h3>{menu.title}</h3>
          <button onClick={() => handleSelectMenu(menu.id)}>Select</button>
          <button onClick={() => handleDeleteMenu(menu.id)}>Delete</button>
        </div>
      ))}
      {selectedMenu && <div>Selected Menu: {selectedMenu.title}</div>}
    </div>
  );
};

export default MenuManager;
