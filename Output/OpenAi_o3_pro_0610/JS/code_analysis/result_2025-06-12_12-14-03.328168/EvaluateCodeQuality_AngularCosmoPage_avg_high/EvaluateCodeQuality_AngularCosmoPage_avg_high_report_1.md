EXTRA SAMPLE – SMALL, MAINTAINABLE SAVE FLOW
================================================================
```js
// page-save.service.js
angular.module('cosmo').service('PageSaveService',
  function (REST, TagService, $q) {

    this.save = function (dto, isDuplicate) {
      const saveOrUpdate = dto.id && !isDuplicate
        ? REST.content.update(dto).$promise
        : REST.content.save(dto).$promise;

      return saveOrUpdate
        .then((saved) => $q.all([
          TagService.replaceTags(saved.id, dto.tags),
          saveRevision(saved.id, dto),
          saveExtras(saved.id, dto.extras)
        ]));
    };

    function saveRevision(id, dto) {
      return REST.contentRevisions.save(
        angular.extend({}, dto, { contentID: id })
      ).$promise;
    }

    function saveExtras(id, extras) {
      const promises = [];
      angular.forEach(extras, (val, key) => {
        promises.push(
          REST.contentExtras.save({
            contentID: id,
            name : key,
            extra: angular.isObject(val) ? angular.toJson(val) : val
          }).$promise
        );
      });
      return $q.all(promises);
    }
});
```
Controller usage becomes a single line:  
```js
PageSaveService.save(vm.page, vm.duplicate)
  .then(() => notify('saved'))
  .catch(() => notify('error', 'alert-error'));
```

The application is now easier to read, test and extend while providing better performance and accessibility.
